                    const [sistolica, diastolica] = lastAssessment.pressao_arterial.split('/').map(Number);

                    if (sistolica < 90 || sistolica > 180 || diastolica < 60 || diastolica > 110) {

                        alerts.push({

                            type: 'warning',

                            message: 'Pressão arterial alterada',

                            action: 'Monitorar e ajustar medicação'

                        });

                    }

                }

               

                if (lastAssessment.saturacao_o2 < 90) {

                    alerts.push({

                        type: 'danger',

                        message: 'Saturação de O2 baixa (<90%)',

                        action: 'Avaliar suporte ventilatório'

                    });

                }

               

                if (lastAssessment.glasgow_total < 13) {

                    alerts.push({

                        type: 'warning',

                        message: 'Glasgow alterado (<13)',

                        action: 'Avaliação neurológica detalhada'

                    });

                }

            }

           

            if (alerts.length === 0) {

                container.innerHTML = '<div class="alert alert-success">Nenhum alerta ativo. Paciente estável.</div>';

                return;

            }

           

            let html = '';

            alerts.forEach(alert => {

                html += `<div class="alert alert-${alert.type}">

                    <strong>${alert.message}</strong><br>

                    <small>Ação recomendada: ${alert.action}</small>

                </div>`;

            });

           

            container.innerHTML = html;

        }
